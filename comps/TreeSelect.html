# TreeSelect

人员/部门树形选择

# 基本用法

<template>
  <TreeSelect
      :option="treeData"
      :isSelectParent="true"
      :isMultiple="true"
      :props="defaultProps"
      :defaultCheckedKeys="defaultCheckedKeys"
      :fetchOption="{
        url: 'https://mock.apifox.cn/m1/1818770-0-default/sps/helpPhone/phone',
        params: { a:1, b:2 },
        option: {},
        callback
      }"
  />
</template>
<script>
export default {
    data () {
        return {
            defaultProps: {
                children: 'children',
                label: 'label',
                value: 'id'
            },
            treeData: [
                {
                    label: '一级 1',
                    id: '1',
                    children: [
                        {
                            label: '二级 1-1',
                            id: '1-1',
                            children: [{
                                label: '三级 1-1-1',
                                id: '1-1-1',
                            }]
                        },
                        {
                            label: '二级 1-2',
                            id: '1-2',
                        }, 
                        {
                            label: '二级 1-3',
                            id: '1-3',
                        }
                    ]
                }, 
                {
                    label: '一级 2',
                    id: '2',
                    children: [
                        {
                            label: '二级 2-1',
                            id: '2-1',
                            children: [{
                                label: '三级 2-1-1',
                                id: '2-1-1',
                            }]
                        }, 
                        {
                            label: '二级 2-2',
                            id: '2-2',
                            children: [{
                                label: '三级 2-2-1',
                                id: '2-2-1',
                            }]
                        }
                    ]
                }
            ],
            defaultCheckedKeys: ['1-2','1-3']
        }
    },
    methods: {
        callback(data) {
            // 对data做某些操作...
            return data
        }
    }
}
</script>

# 属性

参数 说明 类型 可选值 默认值
props 配置选项，具体看下表 object - -
fetchOption 配置选项，具体看下表 object - -
clearable 下拉框是否显示清空icon boolean - true
isSelectParent 是否能选中枝干(非枝末) boolean - true
isMultiple 是否多选 boolean - true
defaultCheckedKeys 默认选中节点([{props.value}, {props.value}, ...]) Array - []
option 选项列表数据(树形结构的对象数组) Array - []
size 下拉框大小 string medium/small/mini -

# props

参数 说明 类型 可选值 默认值
label 指定节点对象的展示名字段 string - 'label'
value 指定节点对象的唯一key string - 'id'
children 指定子树节点对象的属性名 string - 'children'

# fetchOption

参数 说明 类型 可选值 默认值
url 请求地址 string - -
params get请求参数 object - -
option 请求配置 (opens new window) object - -
callback 接受一个数据data, 返回处理后的数据 function - -

# Events

事件名称 说明 回调参数
getValue 选中的节点 共1个参数，选中节点数组
