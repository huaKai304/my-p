# 全局引入element

import Element from 'element-ui';
import 'element-ui/lib/theme-chalk/index.css';
Vue.use(Element)

# 为什么全局引入

  1. 许多通用组件未通过按需导入使用
  2. 减少单个组件打包体积

# 切换私有源

<!-- 添加私有源组 -->
nrm add npm-public http://10.255.33.99:8081/repository/npm-public/

<!-- 切换到组 -->
nrm use npm-public

<!-- 查看当前源 -->
nrm current
or
npm get registry

# 安装

npm install szzt-ui
npm install -D babel-plugin-import

# babel.config.js

const fs = require('fs')
const path = require('path')
const url = (name) => path.resolve(__dirname, `node_modules/${name}/index.css`)
module.exports = {
  presets: [
    '@vue/cli-plugin-babel/preset'
  ],
  plugins: [
    [
      'import',
      {
        libraryName: 'szzt-ui', // 依赖名
        style: (name) => {
          // 判断文件是否存在
          try {
            fs.accessSync(url(name), fs.constants.F_OK)
            return `${name}/index.css`;
          } catch (error) {
            return false
          }
        },
        camel2DashComponentName: false, // 是否需要驼峰转短线
        camel2UnderlineComponentName: false // 是否需要驼峰转下划线
      }
    ]
  ]
}
