# Diy

# 基本用法

<template>
  <Diy :cformProject="cformProject"></Diy>
</template>
<script>
  export default {
    data () {
      return {
        cformProject: {
          dataStruct: {
            fields: [
              {
                type:"INPUT", 
                required: true,
                formItemId: "name",
                inputType:'',//text纯文本，number数字，idCard身份证
                label: "姓名",
                //自定义的增加的值
                attr:'',
                value: "张三",
                placeholder: "",
              },
              {
                type:"textarea",
                formItemId:'address',
                label: "地址",
                placeholder: "", 
              },
              {
                type:"PHONE",
                formItemId:'mobile',
                label: "手机号",
              },
            ]
          }
        }
      }
    }
  }
</script>

# Attributes 属性

参数 说明 类型 可选值 默认值
diyBtnId 部门权限 string - -
cformProject 配置 - - -
originalData 数据 array<originalData> - -
labelWidth label宽度 string - 120px
showDot 显示冒号 boolean - true
maxWidth 表单最大宽度 string - '600px'
disabled 是否不可编辑 boolean - false
disableStar 不可编辑状态是否显示* - - -

# originalData 对象

参数 说明 类型 可选值 默认值
value 表单值 - - -

# cformProject 对象

参数 说明 类型 可选值 默认值
dataStruct 下表 - - -

# dataStruct

参数 说明 类型 可选值 默认值
fields 表单项配置选项 array<fields> - -

# fields

参数 说明 类型 可选值 默认值
type 类型 string 可选值 -
required 是否必填 boolean - false
formItemId 绑定字段 string - -
inputType 输入框类型 string - -
label - string - -
attr - - - -
placeholder 占位 string - -

# type可选值

INPUT / textarea / PHONE / RADIO / CHECKBOX / SELECT / datetime / year / month / week / date / INPUT_MAP / UPLOAD / DEPT_SELECT / SIGN_PAD /
