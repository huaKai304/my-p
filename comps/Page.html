# Page

# 基本用法

<template>
  <page :formConfig="formConfig" :tableConfig="tableConfig" ref="page"></page>
</template>
<script>
  export default {
    data () {
      return {
        formConfig: {
          formConfigArray: [
            {
              cols: [
                {
                  formKey: "name",
                  type: "input",
                  label: "网格名称",
                },
                {
                  type: "btns",
                  handleClick: (type) => {
                    console.log("1", type);
                  },
                  propsConfig: {
                    groupType: "search",
                  },
                },
              ],
            },
          ],
        },
        tableConfig: {
          api: "",
          colsList: [
            {
              label: "测试",
              prop: "name",
            },
            {
              label: "测试2",
              prop: "departmentName",
            },
            {
              label: "switch",
              scopedSlots: {
                default: (scope) => {
                  return (
                    <el-switch
                      value={this.switchValue}
                      active-color={'#13ce66'}
                      onInput={(value) => this.switchValue = value}
                      inactive-color={"#ff4949"}>
                    </el-switch>
                  )
                }
              }
            },
            {
              label: "操作",
              textList: [
                {
                  text: "查看",
                  handleClick: (scope) => {
                    console.log("scope", scope);
                  }
                },
                {
                  text: "删除",
                  color: "red", // 默认为饿了吗的颜色
                  handleClick: (scope) => {
                    console.log("scope1", scope);
                  }
                }
              ],
            }
          ]
        },
      }
    }
  }
</script>

# Attributes 属性

参数 说明 类型 可选值 默认值
formConfig form配置选项, 具体看下表 - - -
tableConfig table配置选项, 具体看下表 - - -
defaultTabActive tabs默认选中值 string - "0"
layout 分页组件属性 string - "total, sizes, prev, pager, next, jumper"
isFirstReq 是否在页面激活时调用分页接口 boolean - true
isShowPagination 是否显示页码 boolean - true
isShowForm 是否渲染筛选栏 boolean - true
isShowTable 是否显示表格 boolean - true
background 分页是否有背景色 boolean - true
isCache 是否缓存 boolean - true
isShowSearch 是否显示筛选栏 boolean - true
