# 基本用法

<template>
  <page :formConfig="formConfig" :tableConfig="tableConfig" ref="page"></page>
</template>
<script>
  export default {
    data () {
      return {
        formConfig: {
          formConfigArray: [
            {
              cols: [
                {
                  formKey: "name",
                  type: "input",
                  label: "网格名称",
                },
                {
                  type: "btns",
                  handleClick: (type) => {
                    console.log("1", type);
                  },
                  propsConfig: {
                    groupType: "search",
                  },
                },
              ],
            },
          ],
        },
        tableConfig: [
          {
            label: "tab1",
            api: "",
            colsList: [
              {
                label: "测试123",
                prop: "name",
              },
              {
                label: "测试2",
                prop: "departmentName",
              },
              {
                label: "switch",
                scopedSlots: {
                  default: (scope) => {
                    return (
                      <el-switch
                        value={this.switchValue}
                        active-color={'#13ce66'}
                        onInput={(value) => this.switchValue = value}
                        inactive-color={"#ff4949"}>
                      </el-switch>
                    )
                  }
                }
              },
              {
                label: "操作",
                textList: [
                  {
                    text: "查看",
                    handleClick: (scope) => {
                      console.log("scope", scope);
                    }
                  },
                  {
                    text: "删除",
                    color: "red", // 默认为饿了吗的颜色
                    handleClick: (scope) => {
                      console.log("scope1", scope);
                    }
                  }
                ],
              }
            ]
          },
          {
            label: "tab2",
            api: "",
            colsList: [
              {
                label: "测试",
                prop: "name",
              },
              {
                label: "测试2",
                prop: "departmentName",
              },
              {
                label: "switch",
                scopedSlots: {
                  default: (scope) => {
                    return (
                      <el-switch
                        value={this.switchValue}
                        active-color={'#13ce66'}
                        onInput={(value) => this.switchValue = value}
                        inactive-color={"#ff4949"}>
                      </el-switch>
                    )
                  }
                }
              },
              {
                label: "操作",
                textList: [
                  {
                    text: "查看",
                    handleClick: (scope) => {
                      console.log("scope", scope);
                    }
                  },
                  {
                    text: "删除",
                    color: "red", // 默认为饿了吗的颜色
                    handleClick: (scope) => {
                      console.log("scope1", scope);
                    }
                  }
                ],
              }
            ]
          },
        ]
      }
    }
  }
</script>

# Attributes 属性

参数 说明 类型 可选值 默认值
formConfig form配置选项, 具体看下表 object - -
tableConfig table配置选项, 具体看下表 - Array<{object}>, object -
defaultTabActive tabs默认选中值 string - "0"
layout 分页组件属性 string - "total, sizes, prev, pager, next, jumper"
isFirstReq 是否在页面激活时调用分页接口 boolean - true
isShowForm 是否渲染筛选栏 boolean - true
isShowSearch 是否显示筛选栏 boolean - true
isShowTable 是否渲染表格 boolean - true
background 是否为分页按钮添加背景色 boolean - true
isCache 是否页面挂载完后调用分页接口 boolean - true

# tableConfig 对象

参数 说明 类型 可选值 默认值
label tableConfig为数组时生效，tabs-plan名称 string - -
colsList 表格列配置选项，具体看下表 Array - -
isShowColSort 是否显示控制列数浮窗 boolean - true
treeProps 渲染嵌套数据的配置选项 object - { hasChildren: 'hasChildren', children: 'children' }
rowKey 同el-table属性rowKey - - -
align 对齐方式 string left/center/right 'left'
minWidth 列最小宽度 string/number - -
spanMethod 合并行或列的计算方法 function({ row, column, rowIndex, columnIndex }) - -
selectionChange event事件，当选择项发生变化时会触发该事件,返回selection function - -
lazy 是否懒加载子节点数据 boolean - -
load 加载子节点数据的函数，lazy为true时生效 function(row, treeNode, resolve) - -
isShowSearch 是否显示筛选栏 boolean - true
isShowPagination 是否显示页码 boolean - true
list 列表数据 Array - []
total 总条目数 number - 0
page 页码 number -
pageSize 每页条数 number - 10
layout 分页组件属性 string - "total, sizes, prev, pager, next, jumper"
requestFn 请求处理函数，优先级高于api function(params, extraConfig) - -
extraConfig 额外参数 any - -
api 全局api的函数名称 function/string - -
handleResult 响应体数据处理函数 function(data) - -
handleParams 额外的请求参数配置 function(params) - -

Table-column Attributes (opens new window)

参数 说明 类型 可选值 默认值

textList 操作栏按钮, 具体看下表 Array<{object}>/object - -
prop 字段名称 string - -
fixed 列是否固定在左侧或者右侧,true 表示固定在左侧 boolean/string true/'left'/'left' 'right'
label 列名称 string - -
scopedSlots
sortable 同Table-column的 sortable - - -
width 列宽度 string/number - '80'
align 对齐方式 string left/center/right 'left'
type 同Table-column的 type - - -
minWidth 列最小宽度 string/number - -
tooltipConfig tooltipConfig配置选项, 具体看下表 - - -
renderHeader 列标题Label区域渲染使用的 Function Function(h, { column, $index }) - -
childCol 自定义展示内容 function - -
