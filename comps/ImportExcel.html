# 基本用法

<template>
  <el-button type="primary" @click="showDialog.open=true">点击打开</el-button>
  <ImportExcel :showDialog="showDialog"></ImportExcel>
</template>
<script>
  export default {
    data () {
      return {
        showDialog: {
          open: false,
          title: '导入',
          downName: '导入模板',
        }
      }
    }
  }
</script>

# option

参数 说明 类型 可选值 默认值
open 是否打开弹窗 boolean - false
title 弹窗名 string - '导入'
downName 下载的文件名 string - '导入模板'
